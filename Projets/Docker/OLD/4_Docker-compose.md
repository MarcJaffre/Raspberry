### A. Dozzle
```yml
################
version: '3.7' #
services:      #
################
#
#
#################################################################
 Dozzle:                                                        #
  # ----------------------------------------------------------- #
  image: 'amir20/dozzle'                                        #
  container_name: 'Dozzle'                                      #
  network_mode: 'bridge'                                        #
  restart: 'unless-stopped'                                     #
  # ----------------------------------------------------------- #
  hostname: 'Dozzle'                                            #
  # ----------------------------------------------------------- #
  deploy:                                                       #
    # --------------------------------------------------------- #
   resources:                                                   #
    limits:                                                     #
     cpus: '0.10'                                               #
     memory: '50M'                                              #
    # --------------------------------------------------------- #
    #reservations:                                              #
    # cpus: '0.01'                                              #
    # memory: '6M'                                              #
  # ----------------------------------------------------------- #
  volumes:                                                      #
   - '/var/run/docker.sock:/var/run/docker.sock'                #
  # ----------------------------------------------------------- #
  ports:                                                        #
   - '1005:8080'                                                #
  # ----------------------------------------------------------- #
  labels:                                                       #
   Cacher: 'Non'                                                #
#################################################################
```
